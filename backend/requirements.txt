# -----------------------------
# FUNÇÕES
# -----------------------------

def calcular_dimensionamento(consumos, tarifa, hsp, performance_ratio, potencia_modulo):
    consumo_medio = sum(consumos) / 12
    geracao_diaria_necessaria = consumo_medio / 30
    potencia_kwp = geracao_diaria_necessaria / (hsp * performance_ratio)
    qtd_modulos = math.ceil((potencia_kwp * 1000) / potencia_modulo)
    potencia_real_kwp = (qtd_modulos * potencia_modulo) / 1000
    geracao_mensal_estimada = potencia_real_kwp * hsp * performance_ratio * 30
    economia_mensal = geracao_mensal_estimada * tarifa
    economia_anual = economia_mensal * 12

    return {
        "consumo_medio": consumo_medio,
        "potencia_kwp": potencia_real_kwp,
        "qtd_modulos": qtd_modulos,
        "geracao_mensal": geracao_mensal_estimada,
        "economia_mensal": economia_mensal,
        "economia_anual": economia_anual
    }

            potencia_real_kwp = (qtd_modulos * modulo_padrao) / 1000

                nome_lead,
                whatsapp,
